### Integrate into your project

Here's an example of how to use `block(){:jsx}`, which creates a reusable component, and `<Block />{:jsx}`, which renders a callback returning JSX.

{/* prettier-ignore */}
```jsx
import { createRoot } from 'react-dom/client';
import { block, Block } from 'million/react';

function Hello({ name }) {
  return <p>Hey there {name}</p>;
}
const HelloBlock = block(Hello);

function App() {
  return <>
    <Block name="Million">{Hello}</Block>
    <HelloBlock name="Million" />
  </>;
}

createRoot(document.getElementById('root')).render(<App />);
```
